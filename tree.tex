\begin{equation}

\begin{equation}
2 \cdot x + 2 \cdot \sin(x) + 2 \cdot x \cdot \cos(x) +  \frac {0 -  \frac {1} {\sqrt{1 - x ^ 2}} } {\arcsin(x) ^ 2} 
\end{equation}
